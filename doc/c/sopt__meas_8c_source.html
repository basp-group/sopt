435 void sopt_meas_fft_r2c_full(void *out, void *in, void **data) {
436 
437  fftw_plan *plan;
438  int *nx, *ny;
439  double complex *y, *y_half;
440  int iu, iv, ind, ind_half;
441  int iu_neg, iv_neg, ind_neg;
442 
443  // Cast intput pointers.
444  y = (double complex*)out;
445  nx = (int*)data[0];
446  ny = (int*)data[1];
447  plan = (fftw_plan*)data[2];
448 
449  // Allocate space for output of real-to-complex FFT before compute
450  // full plane through conjugate symmetry.
451  y_half = (complex double*)malloc((*nx) * (*ny) * sizeof(complex double));
452  SOPT_ERROR_MEM_ALLOC_CHECK(y_half);
453 
454  // Perform real-to-complex FFT.
455  fftw_execute_dft_r2c(*plan,
456  (double*)in,
457  y_half);
458 
459  // Compute other half of complex plane through conjugate symmetry.
460  for (iu = 0; iu < (*nx); iu++) {
461  for (iv = 0; iv < (*ny)/2+1; iv++) {
462 
463  ind_half = iu*((*ny)/2+1) + iv;
464  ind = iu * (*ny) + iv;
465 
466  // Copy current data element.
467  y[ind] = y_half[ind_half];
468 
469  // Compute reflected data through conjugate symmetry if
470  // necessary.
471  if (iu == 0 && iv == 0) {
472  // Do nothing for DC component.
473  }
474  else if (iu == 0) {
475  // Reflect along line iu = 0.
476  iv_neg = (*ny) - iv;
477  ind_neg = iu * (*ny) + iv_neg;
478  if (ind != ind_neg) y[ind_neg] = conj(y_half[ind_half]);
479  }
480  else if (iv == 0) {
481  // Reflect along line iu = 0.
482  iu_neg = (*nx) - iu;
483  ind_neg = iu_neg * (*ny) + iv;
484  if (ind != ind_neg) y[ind_neg] = conj(y_half[ind_half]);
485  }
486  else {
487  // Reflect along diagonal.
488  iv_neg = (*ny) - iv;
489  iu_neg = (*nx) - iu;
490  ind_neg = iu_neg * (*ny) + iv_neg;
491  if (ind != ind_neg) y[ind_neg] = conj(y_half[ind_half]);
492  }
493  }
494  }
495 
496  // Free temporary memory.
497  free(y_half);
498 
499 }
